Задача 1

Напишите Java-программу "Калькулятор", выдающий ошибки при неправильном вводе числа, операции или деления на ноль
Тестовые данные:
  Введите число: 5
  Введите число: 0
  Выберите операцию + - * / : /
Ожидаемый результат :
  Деление на ноль

Тестовые данные:
  Введите число: 5
  Введите число: h
  Выберите операцию + - * / : +
Ожидаемый результат :
  Неправильный ввод

Тестовые данные:
  Введите число: 5
  Введите число: 3
  Выберите операцию + - * / : h
Ожидаемый результат :
   Неправильная операция 

import java.util.InputMismatchException;
import java.util.Scanner;

public class Exercise1{
    public static final String FORMAT = "Результат операции: %d";
    public static final String ENTER_NUMBER = "Введите число: ";

    public static void main(String[] args) {
        int num1 = 0;
        int num2 = 0;
        Scanner scanner = new Scanner(System.in);
        char operation = '0';
        try {
            System.out.print(ENTER_NUMBER);
            num1 = scanner.nextInt();
            System.out.print(ENTER_NUMBER);
            num2 = scanner.nextInt();
            System.out.print("Выберите операцию + - * / : ");
            operation = scanner.next().charAt(0);
        } catch (InputMismatchException e) {
            System.out.println("Неправильный ввод");
            System.exit(0);
        }
        switch (operation) {
            case '+':
                System.out.printf(FORMAT, num1 + num2);
                break;
            case '-':
                System.out.printf(FORMAT, num1 - num2);
                break;
            case '*':
                System.out.printf(FORMAT, num1 * num2);
                break;
            case '/':
                try {
                    System.out.printf(FORMAT, num1 / num2);
                } catch (ArithmeticException e) {
                    System.out.println("Деление на ноль");
                }
                break;
            default:
                System.out.println("Неправильная операция");
                break;
        }
    }
}
