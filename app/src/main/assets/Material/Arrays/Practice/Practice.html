  Массив: [10, -20, 0, 30, 40, 60, 10]

Задача 3

Заполните массив случайным числами от -10 до 10 и выведите максимальное, минимальное и среднее значение.
Пример работы
  Массив: [-4, -6, 6, -9, -6]
Ожидаемый результат :
  -4, -6, 6, -9, -6
  max = 6
  min = -9
  avg = -3.8 

import java.lang.*;
 public class Exercise3 {
	public static void main(String[] args) {

        int minA = -10;
        int maxA = 10;
        double[] array = new double[5];
        for (int i = 0; i < array.length; i++) {
            array[i] = rnd(minA,maxA);
        }

        double max = array[0]; // Массив не должен быть пустым
        double min = array[0];
        double avg = 0;
        for (int i = 0; i < array.length; i++) {
            System.out.print(array[i]+" ");
            if(max < array[i])
                max = array[i];
            if(min > array[i])
                min = array[i];
            avg += array[i]/array.length;
        }
        System.out.println();
        System.out.println("max = " + max);
        System.out.println("min = " + min);
        System.out.println("avg = " + avg);
    }
    public static int rnd(int min, int max)
    {
	max -= min;
	return (int) (Math.random() * ++max) + min;
    }
}

Пример вывода: 

-4, -6, 6, -9, -6
max = 6
min = -9
avg = -3.8

import java.util.Scanner;
public class Exercise4
{
 public static void main( String[] args ) {
        int n = 10;
        int m = 10;

        int[][] arr = new int[n+1][m+1];

        int i = 1, j = 1;
        for ( int k = 0; k < n * m; ++k, ++j ) {

            if ( k != 0 && k % m == 0 ) {
                ++i;
                j = 1;
                System.out.println();
            }

            arr[ i ][ j ] = i * j;
            System.out.format( "%4d", arr[ i ][ j ] );
        }
    }
}
